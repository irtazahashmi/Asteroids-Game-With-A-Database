import java.awt.Graphics2D;
import java.util.Random;

@SuppressWarnings({"PMD.DataflowAnomalyAnalysis", "PMD.MissingSerialVersionUID"})
public class Asteroid extends GameObject {

    private static final double MINIMUM_ROTATION = 0.0075;
    private static final double MAXIMUM_ROTATION = 0.0175;
    private static final double DIFFERENCE_IN_ROTATION = MAXIMUM_ROTATION - MINIMUM_ROTATION;
    private static final double MINIMUM_VELOCITY = 0.8;
    private static final double MAXIMUM_VELOCITY = 2;
    private static final double VELOCITY_DIFFERENCE = MAXIMUM_VELOCITY - MINIMUM_VELOCITY;
    private static final double MINIMUM_DISTANCE = 200.0;
    private static final double MAXIMUM_DISTANCE = GameScene.SCREEN_SIZE / 2.0;
    private static final double DISTANCE_DIFFERENCE = MAXIMUM_DISTANCE - MINIMUM_DISTANCE;
    private static final float SPAWN_UPDATES = 10;

    private transient SizeOfAsteroid size;
    private transient double rotationSpeed;

    /**
     * Asteroid constructor.
     * @param random value
     */
    public Asteroid(Random random) {
        super(calculatePosition(random), calculateVelocity(random),
                SizeOfAsteroid.LargeAsteroid.radius, SizeOfAsteroid.LargeAsteroid.points);
        this.rotationSpeed = -MINIMUM_ROTATION + (random.nextDouble() * DIFFERENCE_IN_ROTATION);
        this.size = SizeOfAsteroid.LargeAsteroid;
    }

    /**
     * The constructor creating new asteroid form a parent.
     * @param parent of asteroid
     * @param size of asteroid
     * @param random of asteroid
     */
    public Asteroid(Asteroid parent, SizeOfAsteroid size, Random random) {
        super(new Vector(parent.position), calculateVelocity(random), size.radius, size.points);
        this.rotationSpeed = MINIMUM_ROTATION + (random.nextDouble() * DIFFERENCE_IN_ROTATION);
        this.size = size;

        for (int i = 0; i < SPAWN_UPDATES; i++) {
            updateGame(null);
        }
    }

    private static Vector calculatePosition(Random random) {
        Vector vec = new Vector(GameScene.SCREEN_SIZE / 2.0, GameScene.SCREEN_SIZE / 2.0);
        return vec.add(new Vector(random.nextDouble() * Math.PI * 2)
                .scaleBy(MINIMUM_DISTANCE + random.nextDouble() * DISTANCE_DIFFERENCE));
    }


    private static Vector calculateVelocity(Random random) {
        return new Vector(random.nextDouble() * Math.PI * 2)
                .scaleBy(MINIMUM_VELOCITY + random.nextDouble() * VELOCITY_DIFFERENCE);
    }


    @Override
    public void updateGame(Game game) {
        super.updateGame(game);
        rotate(rotationSpeed);
    }

    @Override
    public void drawObject(Graphics2D graphics2D, Game game) {
        graphics2D.drawPolygon(size.polygon);
        graphics2D.fillPolygon(size.polygon);
    }

    @Override
    public void collisionHandler(Game game, GameObject other) {
        if (other.getClass() != Asteroid.class) {
            if (size != SizeOfAsteroid.SmallAsteroid) {
                SizeOfAsteroid spawnSize = SizeOfAsteroid.values()[size.ordinal() - 1];

                for (int i = 0; i < 2; i++) {
                    game.registerEntity(new Asteroid(this, spawnSize, game.getRandom()));
                }
            }

            isAlive();

            game.addScore(getPoints());
        }
    }
}
