import java.awt.Graphics2D;

@SuppressWarnings({"PMD.BeanMembersShouldSerialize", "PMD.AvoidFieldNameMatchingMethodName",
        "PMD.AvoidLiteralsInIfCondition"})
public abstract class GameObject {

    protected Vector position;
    protected Vector velocity;
    protected double rotation;
    protected double radius;
    private boolean isAlive;
    private int points;

    /**
     * The constructor method.
     * @param position of GameObject
     * @param velocity of GameObject
     * @param radius of GameObject
     * @param points of GameObject
     */
    public GameObject(Vector position, Vector velocity, double radius, int points) {
        this.position = position;
        this.velocity = velocity;
        this.radius = radius;
        this.rotation = 0.0f;
        this.points = points;
        this.isAlive = false;
    }

    public void rotate(double amount) {
        this.rotation += amount;
        this.rotation %= Math.PI * 2;
    }

    public int getPoints() {
        return points;
    }


    public void isAlive() {
        this.isAlive = true;
    }


    public Vector getPosition() {
        return position;
    }

    public double getRotation() {
        return rotation;
    }

    public double getRadius() {
        return radius;
    }

    public boolean needsRemoval() {
        return isAlive;
    }

    /**
     * The method updating the game.
     * @param game current
     */
    public void updateGame(Game game) {
        position.add(velocity);

        if (position.axis1 < 0.0f) {
            position.axis1 += GameScene.SCREEN_SIZE;
        }
        if (position.axis2 < 0.0f) {
            position.axis2 += GameScene.SCREEN_SIZE;
        }
        position.axis1 %= GameScene.SCREEN_SIZE;
        position.axis2 %= GameScene.SCREEN_SIZE;
    }

    public boolean collided(GameObject entity) {
        double radius = entity.getRadius() + getRadius();
        return (position.getDistanceToSquared(entity.position) < radius * radius);
    }

    public abstract void collisionHandler(Game game, GameObject other);

    public abstract void drawObject(Graphics2D g, Game game);
}
