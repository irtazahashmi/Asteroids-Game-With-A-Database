import java.awt.Graphics2D;

public class Missle extends GameObject {

    private static final double VELOCITY = 7.0;

    private static final int RANGE = 65;

    private int range;

    public Missle(GameObject owner, double angle) {
        super(new Vector(owner.position), new Vector(angle).scaleBy(VELOCITY), 2.0, 0);
        this.range = RANGE;
    }

    @Override
    public void updateGame(Game game) {
        super.updateGame(game);

        this.range--;
        if (range <= 0) {
            isAlive();
        }
    }

    @Override
    public void collisionHandler(Game game, GameObject other) {
        if (other.getClass() != Rocket.class) {
            isAlive();
        }
    }

    @Override
    public void drawObject(Graphics2D graphics2D, Game game) {
        graphics2D.drawOval(-1, -1, 2, 2);
    }

    public int getRange() {
        return range;
    }

    public void setRange(int range) {
        this.range = range;
    }
}
